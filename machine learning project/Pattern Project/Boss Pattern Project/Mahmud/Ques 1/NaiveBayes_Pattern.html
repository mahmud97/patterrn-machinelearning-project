df=pd.read_csv('preference1.csv')



df = df.drop('Unnamed: 0', 1)

df.columns=['cnote','tatt','tint','lint','pref']

#df.set_value(87,'tatt','Low')
train,test=train_test_split(df,test_size=0.1)
df=train
dfO=df.loc[df['pref']=='Online tutorials']
dfC=df.loc[df['pref']=='Class lectures']
o=len(dfO)/len(df)
c=len(dfC)/len(df)
cnoteo={}
cnotec={}
tatto={}
tattc={}
tinto={}
tintc={}
linto={}
lintc={}

cnoteo['High'] = len(dfO.loc[dfO['cnote']=='High'])/len(dfO)
cnoteo['Low'] = len(dfO.loc[dfO['cnote']=='Low'])/len(dfO)
cnoteo['Medium'] = len(dfO.loc[dfO['cnote']=='Medium'])/len(dfO)

cnotec['High'] = len(dfC.loc[dfC['cnote']=='High'])/len(dfC)
cnotec['Low'] = len(dfC.loc[dfC['cnote']=='Low'])/len(dfC)
cnotec['Medium'] = len(dfC.loc[dfC['cnote']=='Medium'])/len(dfC)


tatto['High'] = len(dfO.loc[dfO['tatt']=='High'])/len(dfO)
tatto['Low'] = len(dfO.loc[dfO['tatt']=='Low'])/len(dfO)
tatto['Medium'] = len(dfO.loc[dfO['tatt']=='Medium'])/len(dfO)

tattc['High'] = len(dfC.loc[dfC['tatt']=='High'])/len(dfC)
tattc['Low'] = len(dfC.loc[dfC['tatt']=='Low'])/len(dfC)
tattc['Medium'] = len(dfC.loc[dfC['tatt']=='Medium'])/len(dfC)

tinto['High'] = len(dfO.loc[dfO['tint']=='High'])/len(dfO)
tinto['Low'] = len(dfO.loc[dfO['tint']=='Low'])/len(dfO)
tinto['Medium'] = len(dfO.loc[dfO['tint']=='Medium'])/len(dfO)

tintc['High'] = len(dfC.loc[dfC['tint']=='High'])/len(dfC)
tintc['Low'] = len(dfC.loc[dfC['tint']=='Low'])/len(dfC)
tintc['Medium'] = len(dfC.loc[dfC['tint']=='Medium'])/len(dfC)

linto['High'] = len(dfO.loc[dfO['lint']=='High'])/len(dfO)
linto['Low'] = len(dfO.loc[dfO['lint']=='Low'])/len(dfO)
linto['Medium'] = len(dfO.loc[dfO['lint']=='Medium'])/len(dfO)

lintc['High'] = len(dfC.loc[dfC['lint']=='High'])/len(dfC)
lintc['Low'] = len(dfC.loc[dfC['lint']=='Low'])/len(dfC)
lintc['Medium'] = len(dfC.loc[dfC['lint']=='Medium'])/len(dfC)





correct=0
for index,rows in test.iterrows():
    
    cnote=rows['cnote']
    tatt=rows['tatt']
    tint=rows['tint']
    lint=rows['lint']
    online=cnoteo[cnote]*tatto[tatt]*tinto[tint]*linto[lint]*o
    clas=cnotec[cnote]*tattc[tatt]*tintc[tint]*lintc[lint]*c
    out=''
    
    if(online>clas):
        out='Online tutorials'
    else:
        
        out='Class lectures'
    if(out==rows['pref']):
        correct=correct+1
        
        

    

print('Accuracy:',correct/len(test)*100,'%')
l.append(correct/len(test)*100)

Accuracy: 90.0 %
